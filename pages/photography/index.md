</div>


<!-- Lightbox Interactivity -->
<script>
    const modal = document.getElementById('photoModal');
    const modalImg = document.getElementById('modalImg');
    const photos = document.querySelectorAll('.photo-img');


    // 1. Listen for clicks on every photo in the grid
    photos.forEach(photo => {
        photo.addEventListener('click', function() {
            modal.classList.add('active'); // Show the modal
            modalImg.src = this.src;       // Copy the clicked image's source to the modal
        });
    });


    // 2. Listen for clicks on the dark background to close it
    modal.addEventListener('click', function(e) {
        // Only close if they clicked the background, not the image itself
        if (e.target !== modalImg) {
            modal.classList.remove('active');
            // Optional: clear the src after closing so it doesn't flash the old image next time
            setTimeout(() => { modalImg.src = ''; }, 300);
        }
    });
</script>
